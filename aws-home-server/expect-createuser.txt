#!/usr/bin/expect -f

set force_conservative 0  ;# set to 1 to force conservative mode even if
			  ;# script wasn't run conservatively originally
if {$force_conservative} {
	set send_slow {1 .1}
	proc send {ignore arg} {
		sleep .1
		exp_send -s -- $arg
	}
}


set timeout -1
spawn sudo -u postgres createuser roundcube -P
match_max 100000
expect -exact "Enter password for new role: "
send -- "${PG_PASS}\r"
expect -exact "\r
Enter it again: "
send -- "${PG_PASS}\r"
expect eof
